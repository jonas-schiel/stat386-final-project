for i, row in Final.iterrows():
    if pd.isna(row["President"]):

        current = (row["Year"], month_map[row["Month"]])

        for pres, (sy, sm), (ey, em) in terms:
            start = (sy, month_map[sm])
            end   = (ey, month_map[em])

            if start <= current < end:
                Final.loc[i, "President"] = pres
                break
